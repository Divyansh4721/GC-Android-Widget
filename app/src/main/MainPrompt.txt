foreach ($FilePath in $FilePaths) {
    if (Test-Path -Path $FilePath -PathType Leaf) {
        $AbsolutePath = (Resolve-Path $FilePath).Path
        $FileContent = Get-Content -Path $FilePath -Raw

}
